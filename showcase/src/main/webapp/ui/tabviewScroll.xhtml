TabView - Scrollable

Tabs can be scrolled to deal with huge number of tabs.

Content for #{tab}

Source

<p:tabView scrollable="true" value="\#{tabBean.tabs}" var="tab">
    <p:tab title="\#{tab}">
        <h3>Content for #{tab}</h3>
    </p:tab>
</p:tabView>
		
public class TabBean {

    private List<String> tabs;

    public TabBean() {
        tabs = new ArrayList<String>();

        for(int i = 0; i < 30; i++) {
           tabs.add("Tab " + i); 
        }
    }

    public List<String> getTabs() {
        return tabs;
    }    
}
